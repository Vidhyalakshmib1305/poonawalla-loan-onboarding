# ── 1. GROQ API KEY (Required — free at console.groq.com) ─────────────────
# Without this: Document OCR, LLM extraction, and the AI chatbot will all fail.
# The app will start but every AI step will crash with a 500 error.
GROQ_API_KEY=gsk_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx

# ── 2. SECRET KEY (Required) ──────────────────────────────────────────────
# Any random string — used to sign session tokens.
# Generate one: python -c "import secrets; print(secrets.token_hex(32))"
SECRET_KEY=replace-with-a-random-64-character-string-here

# ── 3. ENVIRONMENT ────────────────────────────────────────────────────────
ENVIRONMENT=development
# Set to "production" when running in Docker

# ── 4. AI MODEL SETTINGS ──────────────────────────────────────────────────
WHISPER_MODEL=base
TF_ENABLE_ONEDNN_OPTS=0

# ── 5. DATABASE ───────────────────────────────────────────────────────────
# Leave blank for local dev (creates audit.db in backend folder)
# Set to /app/data/audit.db when running in Docker

# ── 6. SMTP EMAIL — OTP delivery (Optional but recommended) ───────────────
# Without this: OTP is printed to the backend console AND shown on the login
# screen as "Demo OTP: xxxxxx" — so the app still works for demos.
# With this: OTP is sent to the applicant's email silently (production mode).
#
# How to get Gmail App Password (takes 2 minutes):
#   1. Go to myaccount.google.com
#   2. Security → 2-Step Verification (must be ON first)
#   3. Search "App Passwords" in the search bar
#   4. Name it "Poonawalla OTP" → Generate
#   5. Copy the 16-character password shown (spaces are fine)
SMTP_EMAIL=your.gmail.address@gmail.com
SMTP_PASSWORD=xxxx xxxx xxxx xxxx

# ── 7. OCR PATHS (Windows local dev only) ─────────────────────────────────
# These are auto-detected in Docker (Linux). Only needed on Windows.

# Download Poppler:   https://github.com/oschwartz10612/poppler-windows/releases
